前言

本實作將使用 JavaScript 中的 Fetch 語法去抓取公開資料 API(UBike)的數量，並可以讓使用者輸入關鍵字搜尋對應站點目前的單車數量。

卡住的程式碼


const API =
  "https://tcgbusfs.blob.core.windows.net/dotapp/youbike/v2/youbike_immediate.json";
const inputWord = document.querySelector(".form-control fs-3");
const resultList = document.querySelector("siteList list-group");

async function getRawData(searchBarInput) {
  const getData = await fetch(API);
  const jsonData = await getData.json();
  console.log(jsonData);
}

function searchInputWord() {
  return jsonData.filter.ar((x) => {
    x === "inputWord.value";
  });
}

function putResultToWeb() {
  searchInputWord.forEach((r) => {
    resultList.insertAdjacentElement(r.);
  });
}
