JavaScript 使用 Fetch 實作 API 資料抓取

Posted by     "Mingo" on Sunday, March 27, 2022

本實作練習將使用 JavaScript 中的 Fetch 語法去抓取公開資料 API(UBike)的數量，並可以讓使用者輸入關鍵字搜尋對應站點目前的單車數量。

寫在前面

這次的實作因為語法不夠熟悉，debug 很久，以下紀錄練習時卡住的地方。

1.Fetch 回來的 json 資料要如何處裡 2.forEach 和 map 方式遍歷資料的差異 3.如何使用 filter 方法從 json 檔中篩選出使用者輸入的地名 4.為何要阻止表單(Form)的預設行為，不阻止就送出查詢會怎樣 5.何謂"物件解構”，這樣的寫法有甚麼優點

JS 程式碼

const API =
  "https://tcgbusfs.blob.core.windows.net/dotapp/youbike/v2/youbike_immediate.json";
const resultList = document.querySelector("siteList list-group");

// # 監聽輸入列，若有"submit"的動作，
document.querySelector("#searchForm").addEventListener("submit", function (e) {
  // e為Event Object
  const input = document.querySelector("#searchKeyword").value;
  e.preventDefault(); //阻止預設行為
  checkInput(input); //將輸入列的值作為參數帶入checkInput這個function中
});

// ## 判斷輸入欄的值是否為空白
function checkInput(keywords) {
  if (keywords === "") {
    //若為空白，就跳出提示視窗
    siteListReset(); //呼叫清空搜尋欄function
    alert("輸入關鍵字搜尋路名，例如：八德路");
  } else {
    //若非空白，就呼叫
    siteListReset(); //呼叫siteListReset function 清空搜尋欄
    searchData(keywords); // 將輸入的關鍵字帶入searchData function
  }
}

// 將使用者輸入在搜尋列的內容丟入searchData function函數中
async function searchData(keywords) {
  // 宣告一個非同步函數async function，並搭配await，讓資料取得後再繼續執行
  const youbikeAPI =
    "https://tcgbusfs.blob.core.windows.net/dotapp/youbike/v2/youbike_immediate.json";
  const rawData = await fetch(youbikeAPI);
  // 將資料轉換為json格式
  const data = await rawData.json();
  // 使用filter篩選出符合的資料
  const getFilData = data.filter((filData) => filData.ar.includes(keywords));

  // 使用forEach將資料一一注入頁面中
  getFilData.forEach((dataProperty) => {
    let { ar: address, sna: location, sbi: num } = dataProperty;
    let getDataList = `<li class="list-group-item fs-5">
      <i class="fas fa-bicycle"></i>
       ${location.replace("YouBike2.0_", "")} (${num})<br>
      <small class="text-muted">${address}</small>
      </li>`;
    siteList.insertAdjacentHTML("beforeend", getDataList);
  });
}
